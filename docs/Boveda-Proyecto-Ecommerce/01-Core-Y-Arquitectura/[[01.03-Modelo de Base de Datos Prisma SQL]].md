# [[01.03-Modelo de Base de Datos Prisma SQL]]

Esta nota detalla el esquema relacional completo construido con **Prisma ORM** y **PostgreSQL**. Este modelo cubre todas las necesidades del negocio: usuarios con datos privados para administradores, productos con múltiples imágenes y estados de inventario, reseñas públicas, órdenes con soporte de envío y pago dual, y la tabla crítica de blindaje antifraude para comprobantes de pago manual.

---

## 1. Esquema Completo (`prisma/schema.prisma`)

```prisma
datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

generator client {
  provider = "prisma-client-js"
}

// ==========================================
// 1. MODELO DE USUARIOS Y PERFILES
// ==========================================
model User {
  id            String    @id @default(cuid())
  email         String    @unique
  passwordHash  String
  username      String    @unique // Nombre público para reseñas
  
  // Datos privados (Visibles exclusivamente por el Administrador)
  firstName     String?   
  lastName      String?   
  dni           String?   
  
  phone         String?
  avatarUrl     String?
  address       String?
  role          Role      @default(CUSTOMER)
  
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt
  
  orders        Order[]
  reviews       Review[]
  paymentProofs PaymentProof[]
}

enum Role {
  CUSTOMER
  ADMIN
}

// ==========================================
// 2. MODELO DE PRODUCTOS Y CATÁLOGO
// ==========================================
model Product {
  id            String         @id @default(cuid())
  name          String
  slug          String         @unique
  description   String         @db.Text
  price         Decimal        @db.Decimal(10, 2)
  category      String         // Relacionado con las categorías visuales
  anime         String         // Franquicia o serie de anime
  brand         String         // Fabricante (Good Smile, MegaHouse, etc.)
  status        ProductStatus  // ONLINE, STOCK, PREVENTA, AGOTADO
  expectedDate  DateTime?      // Fecha estimada de llegada (exclusivo para PREVENTA)
  stockQty      Int            @default(0)
  images        String[]       // Galería de imágenes múltiples para la vista detallada
  
  createdAt     DateTime       @default(now())
  updatedAt     DateTime       @updatedAt
  
  reviews       Review[]
  orderItems    OrderItem[]
}

enum ProductStatus {
  ONLINE
  STOCK
  PREVENTA
  AGOTADO
}

// ==========================================
// 3. MODELO DE RESEÑAS Y CALIFICACIONES
// ==========================================
model Review {
  id        String   @id @default(cuid())
  rating    Int      // Calificación de 1 a 5 estrellas
  comment   String   @db.Text
  
  userId    String
  user      User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  
  productId String
  product   Product  @relation(fields: [productId], references: [id], onDelete: Cascade)
  
  createdAt DateTime @default(now())
}

// ==========================================
// 4. MODELO DE ÓRDENES Y TRANSACCIONES
// ==========================================
model Order {
  id              String        @id @default(cuid())
  processCode     String        @unique // Código único de la boleta virtual (Trazabilidad)
  
  userId          String
  user            User          @relation(fields: [userId], references: [id])
  
  totalAmount     Decimal       @db.Decimal(10, 2)
  shippingMethod  ShippingType  // PRESENCIAL, SHALOM, MOTORIZADO
  shippingStatus  String        @default("PENDIENTE_COORDINACION")
  
  paymentMethod   PaymentType   // MERCADO_PAGO, MANUAL_QR
  status          OrderStatus   @default(PENDING_PAYMENT)
  
  createdAt       DateTime      @default(now())
  updatedAt       DateTime      @updatedAt

  items           OrderItem[]
  paymentProof    PaymentProof?
}

enum ShippingType {
  PRESENCIAL
  SHALOM
  MOTORIZADO
}

enum PaymentType {
  MERCADO_PAGO
  MANUAL_QR
}

enum OrderStatus {
  PENDING_PAYMENT
  VERIFYING_MANUAL
  PAID_APPROVED
  REJECTED
  COMPLETED
}

// Detalle de ítems por orden (Tabla intermedia Producto - Orden)
model OrderItem {
  id        String   @id @default(cuid())
  orderId   String
  order     Order    @relation(fields: [orderId], references: [id], onDelete: Cascade)
  
  productId String
  product   Product  @relation(fields: [productId], references: [id])
  
  quantity  Int
  price     Decimal  @db.Decimal(10, 2) // Precio congelado al momento de la compra
}

// ==========================================
// 5. MODELO DE COMPROBANTES DE PAGO MANUAL (POS)
// ==========================================
model PaymentProof {
  id              String      @id @default(cuid())
  
  orderId         String      @unique
  order           Order       @relation(fields: [orderId], references: [id], onDelete: Cascade)
  
  userId          String
  user            User        @relation(fields: [userId], references: [id])
  
  imageUrl        String      // URL de la captura de pantalla del voucher subido
  operationNumber String      // Número de operación ingresado por el usuario
  status          ProofStatus @default(PENDING) // PENDING, APPROVED, REJECTED
  
  createdAt       DateTime    @default(now())
}

enum ProofStatus {
  PENDING
  APPROVED
  REJECTED
}

```

## 2. Explicación de Relaciones y Procesos de Negocio en la Base de Datos

* **Protección de Datos Privados vs Públicos (`User`)**:
* Los campos como `username` y `avatarUrl` son públicos para sustentar el sistema de reseñas.
* Los campos `firstName`, `lastName` y `dni` están aislados en el modelo para que **nunca** se expongan en las APIs públicas del cliente, siendo accesibles únicamente por el rol de `ADMIN`.


* **Inventario Dinámico y Galería (`Product`)**:
* El campo `images` de tipo arreglo de strings (`String[]`) permite almacenar múltiples URLs para alimentar la galería interactiva en la página de detalles del producto.
* El enum `ProductStatus` junto con `expectedDate` gestiona la lógica de artículos en stock inmediato frente a las preventas con fecha de importación.


* **Trazabilidad Antifraude (`Order` y `PaymentProof`)**:
* Cada orden posee un `processCode` único que sirve como comprobante ante cualquier reclamo.
* Si el usuario selecciona el método de pago manual, se crea una relación obligatoria con `PaymentProof`, donde se almacena la captura del voucher y el número de operación para su posterior revisión en la bandeja POS del administrador.

Este es el script SQL completo y definitivo para PostgreSQL, perfectamente alineado con el modelo de Prisma.

Al utilizar **Cloudinary** como hosting de imágenes, la base de datos no almacenará los archivos pesados, sino únicamente las **URLs (en formato de texto)** que devuelve Cloudinary al subir un archivo. Esto está contemplado en los campos `avatarUrl` (para perfiles), `images` (arreglo de URLs para productos) e `imageUrl` (para los comprobantes del POS).

```sql
-- ==========================================
-- 1. CREACIÓN DE ENUMS (Tipos de datos estrictos)
-- ==========================================
CREATE TYPE "Role" AS ENUM ('CUSTOMER', 'ADMIN');
CREATE TYPE "ProductStatus" AS ENUM ('ONLINE', 'STOCK', 'PREVENTA', 'AGOTADO');
CREATE TYPE "ShippingType" AS ENUM ('PRESENCIAL', 'SHALOM', 'MOTORIZADO');
CREATE TYPE "PaymentType" AS ENUM ('MERCADO_PAGO', 'MANUAL_QR');
CREATE TYPE "OrderStatus" AS ENUM ('PENDING_PAYMENT', 'VERIFYING_MANUAL', 'PAID_APPROVED', 'REJECTED', 'COMPLETED');
CREATE TYPE "ProofStatus" AS ENUM ('PENDING', 'APPROVED', 'REJECTED');

-- ==========================================
-- 2. TABLA DE USUARIOS Y PERFILES
-- ==========================================
CREATE TABLE "User" (
    "id" TEXT NOT NULL,
    "email" TEXT NOT NULL,
    "passwordHash" TEXT NOT NULL,
    "username" TEXT NOT NULL,
    
    -- Datos Privados (Exclusivos para visualización del Admin)
    "firstName" TEXT,
    "lastName" TEXT,
    "dni" TEXT,
    
    -- Datos de Contacto y Perfil
    "phone" TEXT,
    "address" TEXT,
    "avatarUrl" TEXT, -- Almacenará la URL generada por Cloudinary para la foto de perfil
    "role" "Role" NOT NULL DEFAULT 'CUSTOMER',
    
    "createdAt" TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "updatedAt" TIMESTAMP(3) NOT NULL,

    CONSTRAINT "User_pkey" PRIMARY KEY ("id")
);

-- ==========================================
-- 3. TABLA DE PRODUCTOS (Inventario / Catálogo)
-- ==========================================
CREATE TABLE "Product" (
    "id" TEXT NOT NULL,
    "name" TEXT NOT NULL,
    "slug" TEXT NOT NULL,
    "description" TEXT NOT NULL,
    "price" DECIMAL(10,2) NOT NULL,
    "category" TEXT NOT NULL,
    "anime" TEXT NOT NULL,
    "brand" TEXT NOT NULL,
    "status" "ProductStatus" NOT NULL,
    "expectedDate" TIMESTAMP(3),
    "stockQty" INTEGER NOT NULL DEFAULT 0,
    "images" TEXT[], -- Arreglo de URLs de Cloudinary para la galería del producto
    
    "createdAt" TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "updatedAt" TIMESTAMP(3) NOT NULL,

    CONSTRAINT "Product_pkey" PRIMARY KEY ("id")
);

-- ==========================================
-- 4. TABLA DE RESEÑAS (Vinculada al perfil público)
-- ==========================================
CREATE TABLE "Review" (
    "id" TEXT NOT NULL,
    "rating" INTEGER NOT NULL,
    "comment" TEXT NOT NULL,
    
    "userId" TEXT NOT NULL,
    "productId" TEXT NOT NULL,
    
    "createdAt" TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT "Review_pkey" PRIMARY KEY ("id")
);

-- ==========================================
-- 5. TABLA DE ÓRDENES (Transacciones y Logística)
-- ==========================================
CREATE TABLE "Order" (
    "id" TEXT NOT NULL,
    "processCode" TEXT NOT NULL, -- Código único de la boleta virtual (Ej: ORD-X8A9B)
    
    "userId" TEXT NOT NULL,
    
    "totalAmount" DECIMAL(10,2) NOT NULL,
    "shippingMethod" "ShippingType" NOT NULL,
    "shippingStatus" TEXT NOT NULL DEFAULT 'PENDING_COORDINACION',
    "paymentMethod" "PaymentType" NOT NULL,
    "status" "OrderStatus" NOT NULL DEFAULT 'PENDING_PAYMENT',
    
    "createdAt" TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "updatedAt" TIMESTAMP(3) NOT NULL,

    CONSTRAINT "Order_pkey" PRIMARY KEY ("id")
);

-- ==========================================
-- 6. TABLA DE ÍTEMS DE LA ORDEN (Detalle de compra)
-- ==========================================
CREATE TABLE "OrderItem" (
    "id" TEXT NOT NULL,
    
    "orderId" TEXT NOT NULL,
    "productId" TEXT NOT NULL,
    
    "quantity" INTEGER NOT NULL,
    "price" DECIMAL(10,2) NOT NULL, -- Congela el precio al momento exacto de la compra

    CONSTRAINT "OrderItem_pkey" PRIMARY KEY ("id")
);

-- ==========================================
-- 7. TABLA DE COMPROBANTES POS (Antifraude / Pago Manual)
-- ==========================================
CREATE TABLE "PaymentProof" (
    "id" TEXT NOT NULL,
    
    "orderId" TEXT NOT NULL,
    "userId" TEXT NOT NULL,
    
    "imageUrl" TEXT NOT NULL, -- Almacenará la URL del voucher/captura subido a Cloudinary
    "operationNumber" TEXT NOT NULL,
    "status" "ProofStatus" NOT NULL DEFAULT 'PENDING',
    
    "createdAt" TIMESTAMP(3) NOT NULL DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT "PaymentProof_pkey" PRIMARY KEY ("id")
);

-- ==========================================
-- 8. ÍNDICES ÚNICOS (Para búsquedas rápidas y evitar duplicados)
-- ==========================================
CREATE UNIQUE INDEX "User_email_key" ON "User"("email");
CREATE UNIQUE INDEX "User_username_key" ON "User"("username");
CREATE UNIQUE INDEX "Product_slug_key" ON "Product"("slug");
CREATE UNIQUE INDEX "Order_processCode_key" ON "Order"("processCode");
CREATE UNIQUE INDEX "PaymentProof_orderId_key" ON "PaymentProof"("orderId");

-- ==========================================
-- 9. LLAVES FORÁNEAS (Relaciones en cascada y restricciones)
-- ==========================================

-- Relaciones de Reseñas
ALTER TABLE "Review" ADD CONSTRAINT "Review_userId_fkey" FOREIGN KEY ("userId") REFERENCES "User"("id") ON DELETE CASCADE ON UPDATE CASCADE;
ALTER TABLE "Review" ADD CONSTRAINT "Review_productId_fkey" FOREIGN KEY ("productId") REFERENCES "Product"("id") ON DELETE CASCADE ON UPDATE CASCADE;

-- Relación de Órdenes
ALTER TABLE "Order" ADD CONSTRAINT "Order_userId_fkey" FOREIGN KEY ("userId") REFERENCES "User"("id") ON DELETE RESTRICT ON UPDATE CASCADE;

-- Relaciones de Ítems de la Orden
ALTER TABLE "OrderItem" ADD CONSTRAINT "OrderItem_orderId_fkey" FOREIGN KEY ("orderId") REFERENCES "Order"("id") ON DELETE CASCADE ON UPDATE CASCADE;
ALTER TABLE "OrderItem" ADD CONSTRAINT "OrderItem_productId_fkey" FOREIGN KEY ("productId") REFERENCES "Product"("id") ON DELETE RESTRICT ON UPDATE CASCADE;

-- Relaciones de Comprobantes POS
ALTER TABLE "PaymentProof" ADD CONSTRAINT "PaymentProof_orderId_fkey" FOREIGN KEY ("orderId") REFERENCES "Order"("id") ON DELETE CASCADE ON UPDATE CASCADE;
ALTER TABLE "PaymentProof" ADD CONSTRAINT "PaymentProof_userId_fkey" FOREIGN KEY ("userId") REFERENCES "User"("id") ON DELETE RESTRICT ON UPDATE CASCADE;

```
